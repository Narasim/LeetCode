class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        
        for i in range(len(nums)):
            sorted = 1
            for k in range(0, len(nums)-1):
                if(nums[k] >= nums[k+1]):
                    temp = nums[k]
                    nums[k] = nums[k+1]
                    nums[k+1] = temp
                    sorted = 0
            if(sorted == 1):
                break
